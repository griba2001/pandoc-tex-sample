(* This are EBNF grammar productions with extra parameters and regular expressions *)

eol = { nl | comment nl | blanks comment nl }

latexSimple = cmdDocClass eol envDocument [nl]

envDocument = cmdBeginEnv("document") eol docContent cmdEndEnv("document") [eol]

cmdBeginEnv(environment) = "\begin{environment}"
cmdEndEnv(environment) = "\end{environment}"

docContent = text | envItemize

(* Bullet list *)

envItemize = cmdBeginEnv("itemize") eol itemizeContent cmdEndEnv("itemize") [eol]

itemizeContent = item { eol item } eol

item = cmdItem text

cmdItem = "\item" [optParam]

optParam = "[" {regexp [^]] | escapedClosingSqBracket} "]"
mandatParam = "{" {regexp [^}]} "}"

escapedClosingSqBracket = "{]}"
escapedOpenSqBracket = "{[}"

text = firstChar {textChar}

textChar = regexp [^\\%\n]

firstChar = escapedOpenSqBracket | textChar
